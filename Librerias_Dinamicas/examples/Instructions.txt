# en la carpeta que contiene hmm.h, hmm.cxx, hmm.i
# 1) instala swig si no está
sudo apt-get update
sudo apt-get install -y swig build-essential python3-dev

# 2) generar wrapper
swig -c++ -python hmm.i

# 3) compilar
g++ -O3 -fPIC -c hmm.cxx -o hmm.o
g++ -O3 -fPIC -c hmm_wrap.cxx -I/usr/include/python3.10 -o hmm_wrap.o   # ajusta include a tu python
g++ -shared hmm.o hmm_wrap.o -o _hmm.so
# ahora en Python puedes: import hmm  (swig genera módulo 'hmm' y submódulo _hmm)
